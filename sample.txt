def isAdditiveNumber(num):
    n = len(num)
    
    for i in range(1, n // 2 + 1):
        for j in range(i + 1, n):
            first = num[:i]
            second = num[i:j]
            
            if (len(first) > 1 and first[0] == '0') or (len(second) > 1 and second[0] == '0'):
                continue  # Numbers with leading zeros are not valid
            
            num1, num2 = int(first), int(second)
            
            if isAdditive(num1, num2, j, num):
                return True
    
    return False

def isAdditive(num1, num2, start, num):
    if start == len(num):
        return True  # End of string, it's an additive sequence
    
    num_sum = num1 + num2
    sum_str = str(num_sum)
    
    if not num.startswith(sum_str, start):
        return False  # The next number doesn't match the sum
    
    return isAdditive(num2, num_sum, start + len(sum_str), num)

# Test cases
input1 = "199100199"
print(input1, "is a valid additive sequence:", isAdditiveNumber(input1))

input2 = "1001"
print(input2, "is a valid additive sequence:", isAdditiveNumber(input2))
